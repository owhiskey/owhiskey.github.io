川大健康每日报自动打卡auto.js脚本

川大健康每日报自动打卡auto.js脚本

十一月 27, 2021

解放双手！实现健康打卡自由，从我做起。

今天偶然看到b站一个up在pc端用python实现了一个自动点赞的脚本，当他说到“脚本可以帮助人们做一些重复又无趣的工作”时，我立马就想到每天恶心的健康打卡。

一天两分钟看着不多，两年每天两分钟可不是个小数字：365*2/60=12.16小时，直接半天没了，以后疫情常态化每天花这时间打卡简直是”谋财害命“。于是我今天花了半天整了个川大微服务自动打卡的脚本，实测能用。

环境和软件具体可以看这个 网址，包括下载在内讲的很详细。（ps：auto.js只能用在安卓）

废话不多说，下面是基于auto.js的代码：（新建个xxx.sj文件，复制进去就行）

auto.waitFor();

setScreenMetrics(1080,2340);

//测试版本，如需改用可微调“swipe”等选项，原理基本相同。具体参考文档：https://hyb1996.github.io/AutoJs-Docs/

//作者：gazer

if(!device.isScreenOn())

{

device.wakeUp();

//这里可以放你的开机密码，这样就可以在应用里设置每日运行了。

}

var appName = “微信”;

app.launchApp(appName); //打开微信

sleep(5000);

if(!text(“四川大学微服务”).findOnce())//如果主页没有微服务的标签，则从公众号打开。

{

click(“通讯录”); //点开通讯录所在位置

sleep(700);

click(“公众号”); //点开公众号所在位置

sleep(700);

click(1050,1491); //点击“s”选项

sleep(700);

}

click(“四川大学微服务”); //点开微服务

sleep(1200);

click(“微服务”);//点开微服务一级菜单

//id(“av9”).className(“android.widget.TextView”).text(“微服务”).findOne().parent().parent().click(); 本句通用性不高故弃用

sleep(4000);

click(“健康每日报”); //健康每日报

sleep(3000);

swipe(500,2100,500,500,800);

click(“请打开手机位置功能”); //获取地理位置

sleep(3000);

var i = 1;

while(i<5)

{

swipe(500,2200,500,100,300);//滑到最底

i++;

}

click(“提交信息”);

sleep(1000);

swipe(500,2330,500,2100,100);

sleep(500);

toast(“今日打卡完成”);

之后在auto.js里设置每日12点自启就ok了！

1月10日补充一段qq自动点赞的代码：

// 测试版本，给tyk做的用于qq自动点赞的脚本

// 官方文档：https://hyb1996.github.io/AutoJs-Docs/

// 作者：gazer

auto.waitFor(); //请求无障碍权限

setScreenMetrics(1080,2340); //设置分辨率为1080*2340

var appName = “QQ”;

var 延迟 = 50; //0.05秒，也就是1秒20个赞。可以改，但建议不要超过屏幕刷新率

function 进入点赞界面()

{

app.launchApp(appName); //打开qq

sleep(2000); //默认两秒的打开时间，可调整

if(!text(“动态”).findOnce()) //用于判断是否处于主界面

{

​ toast(“请确保qq在主界面，脚本已关闭”);

​ exit(); //退出脚本

}

click(108,172); //步骤一点击屏幕位置来实现

sleep(1500); //1.5秒加载时间

click(154,380); //步骤二

sleep(500);

click(995,592); //步骤三

sleep(500);

}

function 赞()

{

className(“ImageView”).desc(“赞”).untilFind().each(function(item){

for(var i = 0; i < 20; i++) //超级会员能点20个赞，nb

{

​ item.click();

​ sleep(延迟);

}

});

}

function 退出()

{

for(var a = 0; a < 3; a++)

{

​ swipe(900,2330,900,2100,100);//模拟全屏手势中“返回上一级”操作，可移除项

​ sleep(200);

}

swipe(500,2330,500,2100,100); //模拟全屏手势中“返回主屏幕”操作，可移除项

sleep(500);

toast(“今日自动点赞完成”);

exit();

}

进入点赞界面();

赞();

退出();
